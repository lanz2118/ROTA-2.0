function init() {
        populateRegionDropdown();
        renderTable(dataset);

        regionDropdown.addEventListener('change', handleRegionChange);
        provinceDropdown.addEventListener('change', handleProvinceChange);
    }

    function populateRegionDropdown() {
        const uniqueRegions = [...new Set(dataset.map(row => row[0]))].sort();
        uniqueRegions.forEach(region => {
            if (region.trim() !== "") {
                const option = document.createElement('option');
                option.value = region;
                option.textContent = region;
                regionDropdown.appendChild(option);
            }
        });
    }

    function populateProvinceDropdown(filteredDataset) {
        provinceDropdown.innerHTML = '<option value="all">-- All Provinces --</option>';
        const uniqueProvinces = [...new Set(filteredDataset.map(row => row[1]))].sort();
        uniqueProvinces.forEach(province => {
            if (province.trim() !== "") {
                const option = document.createElement('option');
                option.value = province;
                option.textContent = province;
                provinceDropdown.appendChild(option);
            }
        });
    }

    function renderTable(dataToDisplay) {
        if (dataToDisplay.length === 0) {
            tableBody.innerHTML = `<tr><td colspan="5" class="no-data">No matching records found.</td></tr>`;
            return;
        }

        let html = '';
        dataToDisplay.forEach((row, index) => {
            const rowId = `details-${index}`;
            // row structural data mapping:
            // 0: Region, 1: Province, 2: Crop, 3: Food Crops, 4: Season

            html += `

                    <td>${row[1]}</td>
                    <td style="font-weight: 600; color: #2c3e50;">${row[2]}</td>
                    <td>${row[3]}</td>
                    <td>${row[4]}</td>

    function toggleDetails(id, rowElement) {
        const detailsRow = document.getElementById(id);
        const isCollapsed = window.getComputedStyle(detailsRow).display === 'none';
        
        if (isCollapsed) {
            detailsRow.style.display = 'table-row';
            rowElement.classList.add('expanded');
            // Change play icon to a small downwards indicator text style if preferred, 
            // but the CSS class 'expanded' handles rotation smoothly via SVG or fonts.
        } else {
            detailsRow.style.display = 'none';
            rowElement.classList.remove('expanded');
        }
    }

    function handleRegionChange() {
        const selectedRegion = regionDropdown.value;
        
        if (selectedRegion === 'all') {
            provinceFilterGroup.style.display = 'none';
            provinceDropdown.value = 'all';
            renderTable(dataset);
        } else {
            const filteredByRegion = dataset.filter(row => row[0] === selectedRegion);
            populateProvinceDropdown(filteredByRegion);
            provinceFilterGroup.style.display = 'block';
            renderTable(filteredByRegion);
        }
    }

    function handleProvinceChange() {
        const selectedRegion = regionDropdown.value;
        const selectedProvince = provinceDropdown.value;

        let filteredData = dataset;

        if (selectedRegion !== 'all') {
            filteredData = filteredData.filter(row => row[0] === selectedRegion);
        }

        if (selectedProvince !== 'all') {
            filteredData = filteredData.filter(row => row[1] === selectedProvince);
        }

        renderTable(filteredData);
    }
